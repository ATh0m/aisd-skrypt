\begin{algorithm}[h]
  \DontPrintSemicolon
  
  \algorithmicrequire{ $a$, $b$ - liczby naturalne}
  
  \algorithmicensure{ $wynik = a \cdot b$ }
  \newline
  
  $ap \leftarrow a$\;
  $bp \leftarrow b$\;
  $wynik \leftarrow 0$\;
  \While{\upshape $ap > 0$}
  {
    \If{\upshape $ap \textsf{ mod } 2 = 1$}
    {
      $wynik \leftarrow wynik + bp$\;
    }
    $ap \leftarrow ap / 2$\;
    $bp \leftarrow bp \cdot 2$\;
  }
  
  \caption{Algorytm rosyjskich wieśniaków}
  \label{alg-wiesniakow}
\end{algorithm}
